import torch
import multiprocessing
import time
from core.engine import AudioEngine
from modules.morph import MorphModule
from modules.harmony import HarmonyModule

# -------------------------
# Detectare hardware auto-adaptiv
# -------------------------
def detect_hardware():
    if torch.cuda.is_available():
        gpu_name = torch.cuda.get_device_name()
        total_vram = torch.cuda.get_device_properties(0).total_memory / (1024**3)
    else:
        gpu_name = "CPU only"
        total_vram = 0
    cpu_cores = multiprocessing.cpu_count()

    # Alegere mod
    if total_vram >= 8:
        mode = "high-end"
        segment_length = 4096
        batch_size = 2
        logging_mode = "full"
    elif total_vram >= 4:
        mode = "full"
        segment_length = 2048
        batch_size = 1
        logging_mode = "full"
    else:
        mode = "light"
        segment_length = 1024
        batch_size = 1
        logging_mode = "minimal"

    return {
        "mode": mode,
        "segment_length": segment_length,
        "batch_size": batch_size,
        "logging": logging_mode,
        "gpu_name": gpu_name,
        "vram_gb": total_vram,
        "cpu_cores": cpu_cores
    }

hw_config = detect_hardware()
print("Hardware detectat:", hw_config)

# -------------------------
# Track demo 30 sec
# -------------------------
track_demo = {
    "audio": [0.1*i for i in range(44100*30)],  # 30 sec dummy audio
    "sample_rate": 44100,
    "metadata": {"history": []}
}

# -------------------------
# Preview 10 sec
# -------------------------
preview_sec = 10
num_samples = preview_sec * track_demo["sample_rate"]
track_preview = {
    "audio": track_demo["audio"][:num_samples],
    "sample_rate": track_demo["sample_rate"],
    "metadata": {"history": []}
}

config_preview = {
    "Morph": {"gender_shift": 0.7, "style_shift": 0.5},
    "Harmony": {"level": 0.5}
}

# -------------------------
# Inițializare engine + module
# -------------------------
engine = AudioEngine()
engine.register_module(MorphModule())
engine.register_module(HarmonyModule())

# -------------------------
# Run preview cu stabilitate
# -------------------------
print("\n=== Running 10 sec preview ===")
try:
    start_time = time.time()
    preview_result = engine.run(track_preview, config_preview)
    end_time = time.time()
    print("Preview complet în {:.2f} sec".format(end_time - start_time))
    print("Preview metadata:", preview_result["metadata"]["history"])
    print("Preview primele 10 samples:", preview_result["audio"][:10])
except Exception as e:
    print("Preview eșuat:", e)
finally:
    if torch.cuda.is_available():
        torch.cuda.empty_cache()

# -------------------------
# Full track processing (opțional, dacă hardware permite)
# -------------------------
if hw_config["mode"] != "light":
    from modules.mastering import MasteringModule
    engine.register_module(MasteringModule())
    print("\n=== Running full track processing ===")
    try:
        start_time = time.time()
        full_result = engine.run(track_demo, config_preview)
        end_time = time.time()
        print("Full track complet în {:.2f} sec".format(end_time - start_time))
        print("Full track metadata:", full_result["metadata"]["history"])
    except Exception as e:
        print("Full track eșuat:", e)
    finally:
        if torch.cuda.is_available():
            torch.cuda.empty_cache()

import torch
import time


class AudioEngine:

    def __init__(self, logging_mode="minimal", device=None):
        self.modules = {}
        self.logging_mode = logging_mode

        if device:
            self.device = device
        else:
            self.device = "cuda" if torch.cuda.is_available() else "cpu"

        self.use_amp = torch.cuda.is_available()

    # -------------------------
    # Register module
    # -------------------------
    def register_module(self, module):
        self.modules[module.name] = module

    # -------------------------
    # Safe run single module
    # -------------------------
    def _run_module(self, module, track, config):

        try:
            torch.cuda.reset_peak_memory_stats() if self.device == "cuda" else None
            start_time = time.time()

            if self.use_amp:
                with torch.cuda.amp.autocast():
                    result = module.process(track, config, device=self.device)
            else:
                result = module.process(track, config, device=self.device)

            end_time = time.time()

            # Profiling
            runtime = end_time - start_time
            vram_peak = 0

            if self.device == "cuda":
                vram_peak = torch.cuda.max_memory_allocated() / (1024 ** 3)

            # Metadata logging
            if "metadata" not in result:
                result["metadata"] = {}

            if "history" not in result["metadata"]:
                result["metadata"]["history"] = []

            result["metadata"]["history"].append(f"{module.name} applied")

            if self.logging_mode == "full":
                result["metadata"]["profiling"] = {
                    "module": module.name,
                    "runtime_sec": round(runtime, 4),
                    "vram_peak_gb": round(vram_peak, 4)
                }

            return result

        except Exception as e:

            if "metadata" not in track:
                track["metadata"] = {"history": []}

            track["metadata"]["history"].append(
                f"{module.name} failed: {str(e)}"
            )

            return track

        finally:
            if self.device == "cuda":
                torch.cuda.empty_cache()

    # -------------------------
    # Chunk processing
    # -------------------------
    def _process_in_chunks(self, module, track, config, chunk_size=44100 * 5):
        """
        Default: 5 sec chunks at 44.1kHz
        """

        audio = torch.tensor(track["audio"], device=self.device)
        chunks = audio.split(chunk_size)

        processed_chunks = []

        for chunk in chunks:
            chunk_track = {
                "audio": chunk,
                "sample_rate": track["sample_rate"],
                "metadata": track.get("metadata", {})
            }

            result = self._run_module(module, chunk_track, config)
            processed_chunks.append(result["audio"])

        full_audio = torch.cat(processed_chunks)

        track["audio"] = full_audio
        return track

    # -------------------------
    # Main run pipeline
    # -------------------------
    def run(self, track, config):

        for module_name, module in self.modules.items():

            module_config = config.get(module_name, {})

            # If module supports chunking
            if hasattr(module, "supports_chunking") and module.supports_chunking:
                track = self._process_in_chunks(module, track, module_config)
            else:
                track = self._run_module(module, track, module_config)

        return track

🧠 Auto-Adaptive Pipeline
Hardware is automatically detected at runtime:

GPU availability

Total VRAM (GB)

CPU core count

Execution mode is selected automatically:

🔹 Light Mode (0–4 GB VRAM or CPU)
Modules: Morph placeholder + Harmony

Segment length: 1024

Batch size: 1

Logging: Minimal

Designed for preview 10s

CPU fallback supported

🔸 Full Mode (4–8 GB VRAM)
Modules: Morph placeholder + Harmony + Mastering

Segment length: 2048–4096

Batch size: 1–2

Logging: Full

🟢 High-End Mode (8+ GB VRAM)
Modules: Morph placeholder + Harmony + Mastering

Segment length: 4096+

Batch size: 2–4

Full logging + profiling

Optimized for long tracks

⚡ 10-Second Preview System
Preview runs only the first 10 seconds of audio.

Purpose:

Fast feedback

Low VRAM usage

Safe parameter testing

Fallbacks implemented for audio < 10 sec and mismatched sample rates.

🔒 Stability & Error Handling (Updated)
✅ Safe Module Execution
try/catch per module

Pipeline continues if a module fails

Errors logged in metadata

✅ Input Validation
Checks for empty audio

Sample rate consistency

NaN / Inf detection

✅ GPU / CPU Fallback
If GPU fails, engine automatically switches to CPU

Ensures process continuity

✅ Cleanup & Memory Management
Automatic torch.cuda.empty_cache() after each module

Pre-allocated buffers for repeated operations

Lazy initialization of heavy kernels

✅ Numeric Stability
Clamp audio values [-1, 1]

Safe divisions with epsilon

Phase/FFT safety in processing

📊 Logging & Metadata (Updated)
History of applied modules

Runtime per module

VRAM peak per module

Metadata safely aggregated across chunks

Fully AMP-compatible logging

Sample metadata structure:

{
  history: [...],
  profiling: {
    module: "Harmony",

Supports supports_chunking flag

Each module receives track + config + device
